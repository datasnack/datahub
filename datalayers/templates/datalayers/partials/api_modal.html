import requests
from urllib.parse import urlencode

import pandas as pd
import matplotlib.pyplot as plt

# set host, like http://localhost:8000, to where your Data Hub is running
HOST = ''
# API TOKEN
TOKEN = ""


def dh_data(query: dict) -> pd.DataFrame:
    response = requests.get(
        f"{HOST}/api/datalayers/data?{urlencode(query)}", headers={"X-API-Key": TOKEN}
    )
    if response.status_code == 200:
        d = response.json()
    else:
        raise Exception(f"Request failed ({response.status_code}): {response.text}")

    df = pd.DataFrame(d["data"])
    df[d["time_col"]] = pd.to_datetime(df[d["time_col"]], format="ISO8601")

    return df
